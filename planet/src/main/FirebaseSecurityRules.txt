// This file contains the security rules of the firebase realtime databse
{
  "rules": {
    // Rules applied to the users tree
    "users": {
      "$uid": {
        ".read": true,
        ".write": "$uid === auth.uid"
      }
    },
    
    // Rules applied to the places tree
    "places": {
      "$placeId":{
        ".read": "root.child('users').child(auth.uid).child('places').child('$placeId').exists() === true",
        ".write": "root.child('users').child(auth.uid).child('places').child('$placeId').exists() === true"
      }
    },
    
    // Rules applied to the events tree
    "events": {
      "$uid": {
        ".read": true,
        ".write": "$uid === auth.uid",
        "$listName": {
          ".indexOn": ["order"]
        }
      }
    }
  }
}
